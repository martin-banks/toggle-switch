Toggle switch

Vanilla JS component to render a toggle switch and run a different function in either state

				
	// Sample usage code
	// Config options
	const toggleOptions = {
	  container: '.tester',
	  callback: {
	    on: ()=> console.log('Button on'),
	    off: ()=> console.log('Button off')
	  },
	  color: {
	    off: false,
	    on: false
	  },
	  size: 'jumbo', // 'small', 'regular', 'large', 'jumbo'
	  theme: 'flat'
	}

	// Instantiate new button
	const myToggle = ToggleSwitch(toggleOptions)

	// A new toggle will render into the target div

	// Get the status of the button
	myToggle.get.active() // true || false
